<script>
(function(){
  var root=document.getElementById('rt');
  if(!root){return;}
  var tabs=root.querySelectorAll('.rt-tab');
  var panels=root.querySelectorAll('.rt-panel');
  function act(i){
    tabs.forEach(function(t){t.classList.toggle('is-on',Number(t.dataset.i)===i);});
    panels.forEach(function(p){p.classList.toggle('is-on',Number(p.dataset.i)===i);});
  }
  tabs.forEach(function(t){
    var i=Number(t.dataset.i);
    t.addEventListener('mouseenter',function(){act(i);});
    t.addEventListener('focus',function(){act(i);});
    t.addEventListener('click',function(){act(i);});
  });
})();
</script>
